import pandas as pd
import numpy as np
from sklearn.preprocessing import LabelEncoder, StandardScaler
from sklearn.model_selection import train_test_split

data_for_model = df[['', '', '', '', '', '']].copy()

#   ( )
cat_features = ['', '']
label_encoders = {}
for col in cat_features:
    le = LabelEncoder()
    data_for_model[col] = le.fit_transform(data_for_model[col])
    label_encoders[col] = le

#   
le_product = LabelEncoder()
data_for_model['_label'] = le_product.fit_transform(data_for_model[''])

# ,  
X = data_for_model[['', '', '', '']]
y_product = data_for_model['_label']      #  : ()
y_amount = data_for_model['']          #  : ()

#  (,   )

models = {
    'RandomForest': RandomForestClassifier(n_estimators=100, random_state=42),
    'XGBoost':      XGBClassifier(n_estimators=100, random_state=42, tree_method='hist', use_label_encoder=False, eval_metric='mlogloss'),
    'LightGBM':     LGBMClassifier(n_estimators=100, random_state=42, verbose=-1),
    'MLP':          MLPClassifier(hidden_layer_sizes=(128, 64, 32), activation='relu', max_iter=500, random_state=42, early_stopping=True, validation_fraction=0.1, verbose=False),
    'LogisticReg':  LogisticRegression(max_iter=1000, solver='lbfgs', multi_class='multinomial', random_state=42, verbose=0)
}

X_train, X_test, y_train, y_test = train_test_split(X_scaled, y_product, test_size=0.2, random_state=42, stratify=y_product)

best_model = None
best_score = 0
topk = 3
results = {}

for name, model in models.items():
    model.fit(X_train, y_train)
    probas = model.predict_proba(X_test)
    # Top-k accuracy  (top-3, top-5)
    score_top3 = top_k_accuracy_score(y_test, probas, k=3)
    score_top5 = top_k_accuracy_score(y_test, probas, k=5)
    #  accuracy(Top-1)
    score_top1 = model.score(X_test, y_test)
    results[name] = {'top1': score_top1, 'top3': score_top3, 'top5': score_top5}
    if score_top5 > best_score:
        best_score = score_top5
        best_model = (name, model)

print("    ( )")
for name, scores in results.items():
    print(f"  - {name}: Top-1 acc = {scores['top1']:.4f}, Top-3 acc = {scores['top3']:.4f}, Top-5 acc = {scores['top5']:.4f}")

def recommend_products(user_info, k=5):
    """
    Input: user_info = dict (, , , )
    Output:   k(, , )
    """
    #  
    sample = {}
    for col in ['', '']:
        val = user_info[col]
        if val in label_encoders[col].classes_:
            sample[col] = label_encoders[col].transform([val])[0]
        else:
            #  0  ( )
            sample[col] = 0
    sample[''] = user_info['']
    sample[''] = user_info['']
    arr = np.array([[sample[''], sample[''], sample[''], sample['']]])
    arr[:,[1,3]] = scaler.transform(arr[:,[1,3]])
    
    #  
    probas = final_cls_model.predict_proba(arr)[0]
    topk_idx = np.argsort(probas)[::-1][:k]
    result = []
    for idx in topk_idx:
        prod_name = le_product.inverse_transform([idx])[0]
        prob = probas[idx]
        #   (  )
        if idx in amount_models:
            pred_amt_log = amount_models[idx].predict(arr)[0]
            pred_amt = np.expm1(pred_amt_log)  #  
            pred_amt = max(pred_amt, 0)  #  
        else:
            pred_amt = df[df['']==prod_name][''].median()
        result.append( (prod_name, prob, int(pred_amt)) )
    return result
